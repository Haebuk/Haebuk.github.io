from airflow.decorators import dag, task
from airflow.models import Variable
from airflow.models.connection import Connection
from airflow.utils.dates import days_ago

default_args = {
    "owner": "airflow",
    "start_date": days_ago(1),
    "depends_on_past": False,
    "catchup": False,
}

@dag(
    default_args=default_args,
    schedule_interval="@once",
    tags=["read_secrets_manager"],
)
def sample_read_secret_manager_dag():
    @task()
    def print_variable_from_sm():
        var = Variable.get("sample_var_1")
        print(var)

    @task()
    def print_connection_from_sm():
        c = Connection(conn_id="sample_connection_1")
        uri = c.get_uri()
        print(uri)

    print_variable_from_sm()
    print_connection_from_sm()

dag = sample_read_secret_manager_dag()
